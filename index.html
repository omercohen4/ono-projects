// ── Recursive Binary Search ─────────
int bsearch(int* arr,
            int lo, int hi,
            int key) {
  if (lo > hi) return -1;
  int mid = lo + (hi - lo) / 2;
  if (arr[mid] == key) return mid;
  return (arr[mid] < key)
    ? bsearch(arr, mid+1, hi,  key)
    : bsearch(arr, lo,  mid-1, key);
}
